```sql
SELECT 
    TO_CHAR(C.NUMRUN, '999g999g999')||'-'||DVRUN AS "RUN_CLIENTE",
    PNOMBRE||' '||SNOMBRE||' '||APPATERNO||' '||APMATERNO AS "NOMBRE_CLIENTE",
    po.nombre_prof_ofic AS "PROFESION/OFICIO",
    to_char(c.fecha_nacimiento, 'DD "de" Month') as "DIA DE CUMPLEAÑOS"
FROM CLIENTE C
    JOIN PROFESION_OFICIO PO ON PO.COD_PROF_OFIC = C.COD_PROF_OFIC
WHERE
    EXTRACT(MONTH FROM c.fecha_nacimiento) > EXTRACT(MONTH FROM SYSDATE)+1
-- GROUP BY c.numrun
--ORDER BY to_char(c.fecha_nacimiento, 'DD "de" Month') 
ORDER BY "DIA DE CUMPLEAÑOS"
;
```

### Caso 3:

- Es necesario agrupar para utilizar SUM.

```sql
SELECT 
    TO_CHAR(CC.fecha_solic_cred, 'MMYYYY') "MES TRANSACCION",
    c.nombre_credito,
    SUM(cc.monto_solicitado),
        CASE
            WHEN SUM(cc.monto_solicitado) BETWEEN 100000 AND 1000000 THEN SUM(cc.monto_solicitado) * 0.01
            WHEN SUM(cc.monto_solicitado) BETWEEN 100001 AND 2000000 THEN SUM(cc.monto_solicitado) * 0.02
            WHEN SUM(cc.monto_solicitado) BETWEEN 200001 AND 4000000 THEN SUM(cc.monto_solicitado) * 0.03
            WHEN SUM(cc.monto_solicitado) BETWEEN 400001 AND 6000000 THEN SUM(cc.monto_solicitado) * 0.04
            WHEN SUM(cc.monto_solicitado) > 6000000 THEN SUM(cc.monto_solicitado) * 0.07
        END
FROM CREDITO_CLIENTE CC
    JOIN CREDITO C ON C.COD_CREDITO = CC.COD_CREDITO
GROUP BY 
cc.fecha_solic_cred, 
c.nombre_credito
;
```
