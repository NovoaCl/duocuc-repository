```SQL
SELECT
    C.NRO_CLIENTE,
    C.NUMRUN,
    C.PNOMBRE,
    COALESCE(CC.FECHA_SOLIC_CRED, SYSDATE),
    NVL(CC.MONTO_SOLICITADO, 0)
FROM CLIENTE C
    LEFT JOIN CREDITO_CLIENTE CC ON C.NRO_CLIENTE = CC.NRO_CLIENTE
;
```

```sql
SELECT
    C.NRO_CLIENTE,
    C.NUMRUN,
    C.PNOMBRE,
    CC.FECHA_SOLIC_CRED,
    CC.MONTO_SOLICITADO,
    CASE 
        WHEN CC.MONTO_SOLICITADO >= 6000000 THEN 'MUY ENDEUDADO'
        WHEN CC.MONTO_SOLICITADO BETWEEN 4000000 AND 6000000 THEN 'ENDEUDADO'
        WHEN CC.MONTO_SOLICITADO > 2000000 AND CC.MONTO_SOLICITADO < 6000000 THEN 'POCO ENDEUDADO'
        ELSE
            'NO CLASIFICADO'
    END
FROM CLIENTE C
    LEFT JOIN CREDITO_CLIENTE CC ON C.NRO_CLIENTE = CC.NRO_CLIENTE
;
```
